如何写SPJ？

第一个参数是标准输入，第二个是标准输出，第三个是选手输出。

程序返回0代表AC，否则如果返回值<=5且大于0，则是WA，错误信息直接输出到屏幕即可

下面是一个范例，标准输出随机0 or 1

#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <string>
#include <fstream>
#include <iostream>
#include <algorithm>
#include <cmath>
using std::string;
using std::ifstream;
using std::min;
using std::cout;
using std::endl;
int main(int argc, char* argv[])
{
  int a = time(0);
  while(a==time(0));
  ifstream std(argv[2]);
  ifstream out(argv[3]);
  char xx;
  char x;
  std>>xx;
  srand(a);
  out>>x;
  int y = x - '0';
  if (y != rand()%2) return 1;
  return 0;
}
